import React, {useState} from 'react';
import { useForm, Controller } from 'react-hook-form';
import {MuiPickersUtilsProvider, DatePicker} from '@material-ui/pickers';
import input from '@material-ui/core/Input';
import 'react-datepicker/dist/react-datepicker.css';
import DateFnsUtils from '@date-io/date-fns';

export default function FavouriteMovies() {
    const {register, handleSubmit, control, formState: {errors} } = useForm();
    // const [startDate, setStartDate] = useState(new Date());

    const [selectedDate, setSelectedDate] = useState(null);

    const handleDateChange = (date) => {
    setSelectedDate(date);
    };

    function ConsoleDate() {
        var date = {
            "Date": document.getElementById("DateOfBirth2").value,
            "Date2": document.getElementById("DateOfBirth").value
        }
        console.log(date);
    }

    return(
        <div id="favourite">
            <form onSubmit={handleSubmit(ConsoleDate)}>
            <MuiPickersUtilsProvider utils={DateFnsUtils}>
                <DatePicker
                    animateYearScrolling
                    disableFuture
                    minDate="1930-01-01"
                    clearable
                    name="dateofbirth"
                    margin="normal"
                    id="DateOfBirth2"
                    placeholder="Date Of Birth"
                    format="yyyy-MM-dd"
                    value={selectedDate}
                    onChange={handleDateChange}
                    KeyboardButtonProps={{
                        'aria-label': 'change date',
                    }}
                    {...register("DateOfBirth", { required: true })}
                />
                {/* Error message when the user did not provide password value in the password field */}
                {errors?.DateOfBirth?.type === "required" && <p className="errormssg">This field is required</p>}
            </MuiPickersUtilsProvider>
            <input type="date" placeholder="Date of Birth" name="dateofbirth" id="DateOfBirth"/>
                <button type="submit">Submit</button>  
            </form>
        </div>
    )
}
